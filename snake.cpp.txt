struct node

    char name[20];

    struct node* next;
};

struct qnode
{
    char name[20];
    int seat,ticket,date;
    struct qnode* qnext;
};


void enqueue(struct qnode **first,int d,char name[20],int date)
{
    struct qnode* n;
    n=*first;
    if(n!=NULL)
    {
        struct qnode* newnode = (struct qnode*)malloc(sizeof(struct qnode));
        newnode->ticket=d;
        newnode->date=date;
        strcpy(newnode->name,name);

        while((n->qnext)!=NULL)
        {
            n=n->qnext;
        }
        n->qnext=newnode;
        newnode->qnext=NULL;
    }
    else if(n==NULL)
    {
        struct qnode* head = (struct qnode*)malloc(sizeof(struct qnode)) ;

        head->ticket=d;
        head->date=date;
        strcpy(head->name,name);

        head->qnext=*first;
        *first=head;
    }
}

struct qnode* dequeue(struct qnode** first)
{
        struct qnode* m;
        m = *first;

        if(m==NULL)
        {
        return NULL;
        }
        else if(m->qnext==NULL)
        {
            *first=NULL;
            return m;
        }
        else
      {
        *first=m->qnext;
        return m;
    }
}
